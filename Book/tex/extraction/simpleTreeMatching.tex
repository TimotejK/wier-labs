{\bf Algorithm:} SimpleTreeMatching($T'$, $T''$)

\begin{algorithm}[H]
	\SetEndCharOfAlgoLine{}
	\SetAlgoNoLine
	% \KwData{this text}
	% \KwResult{how to write algorithm with \LaTeX2e }
	
	\eIf{$T'$ {\rm has the same label of} $T''$}{
		m $\leftarrow$ d($T'$)\;
		n $\leftarrow$ d($T''$)\;
		\For{i = 0 {\rm to} m}{
			$M[i][0] \leftarrow 0$\;		
		}
		\For{j = 0 {\rm to} n}{
			$M[0][j] \leftarrow 0$\;		
		}
		\ForAll{i {\rm such that} $1 \le i \le m$}{
			\ForAll{j {\rm such that} $1 \le i \le n$}{
				$W_{ij}$ = {\rm SimpleTreeMatching($T'(i-1), T''(j-1)$)}\;
				{\rm $M[i][j]$ $\leftarrow$ Max($M[i][j-1]$, $M[i-1][j]$, $M[i-1][j-1]$ + $W_{ij}$)}\;
			}
		}
		return $M[m][n]+1$
	}{
		return 0\;
	}	
\end{algorithm}

\end{LARGE}
